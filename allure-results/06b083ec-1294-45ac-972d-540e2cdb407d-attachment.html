    var validator = $("form.form-body").validate({
        rules: {
            'txtUsername': {

                notEmpty: true
            }
        },
        messages: {

    localStorage.clear();

    function calculateUserTimeZoneOffset() {
        var myDate = new Date();
        var offset = (-1) * myDate.getTimezoneOffset() / 60;
        return offset;
    }

    function submitLoginForm(event, csrfTokenUpdated) {

        var formValid = validator.valid();
        if (formValid) {
            if (csrfTokenUpdated) {
                return true;
            }
            // If csrf is invalid, make AJAX call to fetch csrf token and abort this submit by returning false
            if (!getCsrfToken()) {
                $.post(csrfUrl, {},function(token) {
                    $('#login__csrf_token').val(token);
                    $('#frmLogin').trigger('submit', [true]);
                });
                return false;
            }
            removeCSRFToken();
            return true;
        }
        return false;
    }

    function getCsrfToken() {
        var csrfToken = localStorage.getItem('login_csrf');
        try {
            var now = new Date();
            var csrfTokenObj = JSON.parse(csrfToken);

            if (csrfTokenObj &amp;&amp; csrfTokenObj.expire &amp;&amp; csrfTokenObj.expire &gt; now.getTime()) {
                $('#login__csrf_token').val(csrfTokenObj.token);
                return true;
            }
        } catch (e) {
            removeCSRFToken();
        }
        return false;
    }

    function removeCSRFToken() {
        localStorage.removeItem('login_csrf');
    }

    function saveCSRFToken(token) {
        var now = new Date();
        var expireInMS = now.getTime() + 2 * 60 * 1000;
        localStorage.setItem('login_csrf', JSON.stringify({token: token, expire: expireInMS}));
    }
    
    $(document).ready(function() {
        /* Set cookie with csrf token to expire in 2 minutes */
        saveCSRFToken($('#login__csrf_token').val());

        $('#frmLogin').submit(submitLoginForm);
    });

    }
